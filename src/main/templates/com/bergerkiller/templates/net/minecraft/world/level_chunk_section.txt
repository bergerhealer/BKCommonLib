package net.minecraft.world.level.chunk;

import net.minecraft.world.level.block.state.BlockState;

import com.bergerkiller.bukkit.common.wrappers.BlockData;

import com.bergerkiller.generated.net.minecraft.world.level.chunk.PalettedContainerHandle;

class LevelChunkSection {
#if version >= 1.18
    public boolean isEmpty:hasOnlyAir();
#elseif version >= 1.14
    public boolean isEmpty:c();
#else
    public boolean isEmpty:a();
#endif

#if version >= 1.18
    public (PalettedContainerHandle) PalettedContainer<BlockState> getBlockPalette:getStates();
#elseif version >= 1.13
    public (PalettedContainerHandle) PalettedContainer<BlockState> getBlockPalette:getBlocks();
#elseif version >= 1.9
    public (PalettedContainerHandle) PalettedContainer getBlockPalette:getBlocks();
#else
    // For MC 1.8.8 we do some conversion magic to expose the same API
    // PalettedContainer doesn't actually exist, it is emulated by BKCommonLib
    public (PalettedContainerHandle) char[] getBlockPalette:getIdArray();
#endif

#if version >= 1.18
    public (BlockData) BlockState getBlockData:getBlockState(int x, int y, int z);
#else
    public (BlockData) BlockState getBlockData:getType(int x, int y, int z);
#endif

#if version >= 1.18
    public (void) BlockState setBlockData:setBlockState(int x, int y, int z, (BlockData) BlockState data);
#elseif version >= 1.14
    public (void) BlockState setBlockData:setType(int x, int y, int z, (BlockData) BlockState data);
#else
    public void setBlockData:setType(int x, int y, int z, (BlockData) BlockState data);
#endif

    public void setBlockDataAtBlock(org.bukkit.block.Block block, (BlockData) BlockState data) {
#if version >= 1.18
        instance.setBlockState(block.getX() & 0xf, block.getY() & 0xf, block.getZ() & 0xf, data);
#else
        instance.setType(block.getX() & 0xf, block.getY() & 0xf, block.getZ() & 0xf, data);
#endif
    }
}

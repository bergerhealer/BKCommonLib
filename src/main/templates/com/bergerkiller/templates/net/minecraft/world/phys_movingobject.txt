package net.minecraft.world.phys;

import net.minecraft.core.Direction;

class HitResult {
#if version >= 1.17
    public readonly (org.bukkit.util.Vector) Vec3 pos:location;
#else
    public readonly (org.bukkit.util.Vector) Vec3 pos;
#endif

    public (org.bukkit.block.BlockFace) Direction getDirection() {
#if version >= 1.14
        if (instance instanceof BlockHitResult) {
            return ((BlockHitResult) instance).getDirection();
        } else {
            return Direction.DOWN;
        }
#else
        return instance.direction;
#endif
    }
}

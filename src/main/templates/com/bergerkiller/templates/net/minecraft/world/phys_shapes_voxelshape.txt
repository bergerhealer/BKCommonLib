    public static (Object) VoxelShape createRawFromAABB((Object) List<AABB> handles) {
        if (handles.isEmpty()) {
            return Shapes.empty();
        } else {
            VoxelShape shape = null;
            java.util.Iterator iter = handles.iterator();
            while (iter.hasNext()) {
                AABB aabb = (AABB) iter.next();
                VoxelShape aabb_shape = Shapes.create(aabb);
                if (shape == null) {
                    shape = aabb_shape;
                } else {
                    shape = Shapes.or(shape, aabb_shape); // OR
                }
            }
            return shape;
        }
    }

        return Shapes.create(aabb);

        java.util.List a_cubes = a.getCubes();
        java.util.List b_cubes = b.getCubes();
        if (a_cubes.isEmpty()) {
            return b;
        } else if (b_cubes.isEmpty()) {
            return a;
        }
        int i = 0;
        java.util.Iterator iter;
        Object[] arr = new Object[a_cubes.size() + b_cubes.size()];
        for (iter = a_cubes.iterator(); iter.hasNext(); ) {
            arr[i++] = iter.next();
        }
        for (iter = b_cubes.iterator(); iter.hasNext(); ) {
            arr[i++] = iter.next();
        }
        return com.bergerkiller.bukkit.common.internal.proxy.VoxelShapeProxy.fromAABBHandles(java.util.Arrays.asList(arr));
    }
